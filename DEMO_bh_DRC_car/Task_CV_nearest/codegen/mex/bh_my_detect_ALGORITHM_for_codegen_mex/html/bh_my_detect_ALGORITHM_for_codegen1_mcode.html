1function [BLUE_s, YELL_s, PURP_s] = bh_my_detect_ALGORITHM_for_codegen(I, tf_do_init, marker_s) %#codegen
  2% ALLOWED USAGE:
  3%  >> [BLUE_s, YELL_s, PURP_s] = bh_my_detect_ALGORITHM_for_codegen(I, tf_do_init, marker_s)
  4
  5persistent OBJ_ybp
  6persistent fr_count
  7persistent hax
  8persistent hi
  9
 10if(isempty(fr_count))
 11    fr_count = 0;
 12end
 13
 14% *************************************************************************
 15%      do OBJECT INITIALIZATION
 16% *************************************************************************
 17if(true==tf_do_init || isempty(OBJ_ybp) )
 18    fr_count = 0;
 19    
 20    % create an instance of the ybp finder
 21    OBJ_ybp  = ybp_finder_CLS(marker_s);
 22    
 23end % if(true==tf_do_init)
 24
 25% update the frame counter
 26fr_count = fr_count + 1;
 27
 28% *************************************************************************
 29%      CLASSIFY stuff
 30% *************************************************************************
 31OBJ_ybp = OBJ_ybp.classify(I);
 32
 33[ BLUE_s, YELL_s, PURP_s ] = OBJ_ybp.find_all_stuff();
 34
 35% *************************************************************************
 36%      VISUALIZE  stuff
 37% *************************************************************************
 38
 39if(1 ==coder.target('MATLAB') )
 40
 41    % close any open figures that we want to target
 42    if(true==tf_do_init)
 43       hf = findobj('Type', 'figure', 'Tag', 'TAG_FIG_BH_MY_CV');
 44      if( ~isempty(hf) )
 45          close(hf);
 46      end
 47    end
 48    
 49    num_ROWS = size(I,1);
 50    num_COLS = size(I,2);
 51       
 52    BW = BLUE_s.BW | YELL_s.BW | PURP_s.BW;
 53    b_pix = find(true==BLUE_s.BW);
 54    y_pix = find(true==YELL_s.BW);
 55    p_pix = find(true==PURP_s.BW);
 56    
 57    RGB_mat  = bh_create_byp_RGB_mat(  num_ROWS, num_COLS, ...
 58                                          b_pix,  ...
 59                                          y_pix,  ...
 60                                          p_pix);
 61    
 62    hf = findobj('Type', 'figure', 'Tag', 'TAG_FIG_BH_MY_CV');
 63    if( isempty(hf) )
 64        figure('Tag', 'TAG_FIG_BH_MY_CV', 'Name', 'FIG_BH_MY_CV');
 65        THE_hax(1) = subplot(2,2,1, 'Tag', 'TAG_HX_1');
 66        THE_hax(2) = subplot(2,2,2, 'Tag', 'TAG_HX_2');
 67        THE_hax(3) = subplot(2,2,3, 'Tag', 'TAG_HX_3_TGT_FOR_LINE_TRACK');
 68        THE_hax(4) = subplot(2,2,4, 'Tag', 'TAG_HX_4'); 
 69        hax = THE_hax;
 70    end
 71        
 72    % plot stuff - the if-else makes the plotting faster ! (DN tip)
 73    if isempty(get(hax(1),'Children'))
 74         THE_hi(1) = imshow(I,       'Parent', hax(1));
 75         THE_hi(2) = imshow(RGB_mat, 'Parent', hax(2));
 76         THE_hi(3) = imshow(RGB_mat, 'Parent', hax(3));
 77         THE_hi(4) = imshow(BW,      'Parent', hax(4));   
 78         
 79         % and of course the TAGs will be cleared .. so reassign
 80         hax(3).Tag = 'TAG_HX_3_TGT_FOR_LINE_TRACK';
 81         
 82         hi = THE_hi;
 83    else
 84         hi(1).CData = I;
 85         hi(2).CData = RGB_mat;
 86         hi(3).CData = RGB_mat;
 87         hi(4).CData = BW;
 88    end
 89    
 90    %title( hax(1), sprintf('FRAME = %3d',fr_count) )
 91
 92    drawnow
 93end
 94
 95
 96end % bh_my_ALGORITHM_for_codegen(I)
 97%-#########################################################################
 98% E N D    O F    M A I N    F U N C T I O N
 99%-#########################################################################
100
