1 classdef box_info_CLS %#codegen
  2% ATTENTION:
  3% I want this class to be MATLAB CODER friendly, hence I have prepopulated
  4% some properties with NaNs
  5
  6properties (SetAccess = protected)
  7    BW             ;
  8    pixIND_list    ;
  9    NUM_ROWS    = 0; 
 10    NUM_COLS    = 0;
 11    %---------------------------------
 12    left_edge_xy_col  = [0;0]; 
 13    right_edge_xy_col = [0;0]; 
 14    %---------------------------------
 15    centroid_xy_col   = [0;0];
 16    %---------------------------------
 17    tf_is_detected   = false;
 18end
 19%_#########################################################################
 20methods
 21function OBJ = box_info_CLS(BW, tf_found)
 22      OBJ.BW            = BW;
 23      
 24      OBJ.NUM_ROWS       = size(BW,1);
 25      OBJ.NUM_COLS       = size(BW,2);
 26      
 27      OBJ.tf_is_detected = tf_found;
 28      
 29      OBJ = OBJ.calc_info();
 30end

 31%--------------------------------------------------------------------------  
 32function tf_detected = is_detected(OBJ)   
 33         tf_detected = OBJ.tf_is_detected;
 34end % is_detected()
 35%-------------------------------------------------------------------------- 

 36function OBJ = calc_info(OBJ)
 37   if( ~OBJ.is_detected() ) 
 38       return
 39   end
 40       
 41   % OK: if we're here we have a box
 42   the_size = [OBJ.NUM_ROWS, OBJ.NUM_COLS];
 43   the_ind  = find(OBJ.BW);
 44   [y,x]    = ind2sub(the_size,the_ind); % convert INDS to SUBSCRIPTS
 45      
 46   % the LEFT edge will have the smallest X-co-ord
 47   the_left_edge_x  = 1;
 48   [~,tmp_I]        = min(x);
 49   the_left_edge_x  =  x(tmp_I(1));        % MATLAB CODER friendly
 50   tmp_II           = find(x==the_left_edge_x);
 51   yI               = y(tmp_II);
 52   the_left_edge_y  = min(yI);
 53   
 54   % the RIGHT edge will have the biggest X-co-ord
 55   the_right_edge_x  = 1;
 56   [~,tmp_I]         = max(x);
 57   the_right_edge_x  =  x(tmp_I(1));        % MATLAB CODER friendly
 58   tmp_RR            = find(x==the_right_edge_x);
 59   yR                = y(tmp_RR);
 60   the_right_edge_y  = min(yR);
 61   
 62   % calc the centroid
 63   xc = mean(x);
 64   yc = mean(y);
 65   
 66   % take care of the OUTPUTS
 67   OBJ.left_edge_xy_col   = [the_left_edge_x(1);  the_left_edge_y(1)   ]; % MATLAB CODER friendly
 68   OBJ.right_edge_xy_col  = [the_right_edge_x(1); the_right_edge_y(1)  ]; % MATLAB CODER friendly
 69  
 70   OBJ.centroid_xy_col    = [xc(1); yc(1) ]; % MATLAB CODER friendly
 71
 72end % calc_info()
 73%--------------------------------------------------------------------------  
 74function xy = get_xy(OBJ)
 75    coder.varsize('xy', [2 1]);
 76
 77    xy = [];
 78    if( ~OBJ.is_detected() ) 
 79       return
 80    end
 81   
 82    xy = OBJ.centroid_xy_col;
 83end % get_xy()
 84%--------------------------------------------------------------------------  
 85function my_S = convert_to_struct(OBJ)
 86  % ATTENTION(R2017b):
 87  % the following usage is NOT supported by the MATLAB Coder:
 88  %   >> struct(OBJ)

 92    my_S.NUM_COLS          = OBJ.NUM_COLS;
 93    my_S.left_edge_xy_col  = OBJ.left_edge_xy_col;
 94    my_S.right_edge_xy_col = OBJ.right_edge_xy_col;
 95    my_S.centroid_xy_col   = OBJ.centroid_xy_col;
 96    my_S.tf_is_detected    = OBJ.tf_is_detected; 
 97end
 98%--------------------------------------------------------------------------  
 99end % methods()
