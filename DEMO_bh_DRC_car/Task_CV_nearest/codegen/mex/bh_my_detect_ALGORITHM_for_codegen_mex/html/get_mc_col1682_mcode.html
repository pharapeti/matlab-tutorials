 57function  OBJ = pred_and_correct(OBJ, BW, tf_found)
 58    
 59    % use KF to PREDICT
 60    OBJ.KF     = OBJ.KF.predict();
 61
 62    % now do we have a measurement to allow a CORRECT
 63    lane_O = lane_info_CLS(BW, tf_found);
 64    if( lane_O.is_detected() )
 65        lane_O = lane_O.calc_info();
 66    
 67        [m,c]  = lane_O.get_mc();
 68        
 69        y_meas = [m;c];
 70        
 71        OBJ.KF = OBJ.KF.correct(y_meas);
 72    end
 73   
 74    OBJ.mc_col    = OBJ.KF.x;
 75    %OBJ.x_col_mat = [OBJ.x_col_mat, OBJ.mc_col];
 76    
 77    % and store the size of the image we're processing
 78    OBJ.img_num_rows = size(BW,1);
 79    OBJ.img_num_cols = size(BW,2);
 80end % pred_and_correct()

 81%--------------------------------------------------------------------------

 84end

 86function plot(OBJ, hax, linspec)
 87% ALLOWED usage:
 88%  >> OBJ.plot(hax)
 89%  >> OBJ.plot(hax,'.r')
 90
 91    if(1 ==coder.target('MATLAB') )
 92        if( 2==nargin )
 93            linspec = '.r-';
 94        end
 95
 96        mc_col = OBJ.mc_col;
 97        if( isempty(mc_col) )
 98            return
 99        end
100
101        m = mc_col(1);
102        c = mc_col(2);
103        % OK: we have some m,c data
104        %  y = m.x + c --->  x = (y-c)/m
105        % I'm interested in 2 data points:
106        %  a.) when y = 120, what is x ?
107        %  b.) when y = 240, what is x ?
108
109        y_HALF = OBJ.img_num_rows/2;
110        y_MAX  = OBJ.img_num_rows;
111
112        P1 = [(y_HALF-c)/m ; y_HALF];
113        P2 = [(y_MAX-c)/m ;  y_MAX];
114
115        x  = [P1(1); P2(1)];
116        y  = [P1(2); P2(2)];
117
118        TAG = OBJ.plot_TAG;
119        hL = findall(hax,'Tag',TAG);
120
121        if( isempty(hL) )
122           hold(hax,'on');
123           plot(hax,x,y,linspec,'MarkerSize',22, 'Tag',TAG);
124        else
125           hL.XData = x;
126           hL.YData = y;
127        end
128    end % if(1 ==coder.target('MATLAB') )
129    
130end % plot()
