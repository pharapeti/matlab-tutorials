1classdef my_KF_general_CLS
  2% MY_KF_GENERAL_CLS - a vanilla implementation of a Kalman Filter
  3%--------------------------------------------------------------------------
  4% REF: 
  5%  1.) https://cse.sc.edu/~terejanu/files/tutorialKF.pdf
  6%      - "Discrete Time Kalman Filter tutorial" - Gabriel A. Terejanu
  7%  2.) https://au.mathworks.com/videos/series/understanding-kalman-filters.html
  8%--------------------------------------------------------------------------
  9% Discrete plant:
 10%     x[k] = A.x[k-1]  +  B.u[k-1]   +  w[k-1]
 11%     y[k] = C.x[k]                  +  v[k]
 12%
 13%   where:
 14%           Q = cov(w) ==PROCESS_NOISE covariance matrix
 15%           R = cov(v) ==MEASUREMENT_NOISE covariance matrix
 16%--------------------------------------------------------------------------
 17% PREDICT:
 18%    xhat[k|k-1] = A.x[k-1|k-l]       +  B.u[k-1]
 19%       P[k|k-1] = A.P[k-1|k-1].(A')  +  Q
 20% CORRECT:
 21%       K[k]     = P[k|k-1].(C'). inv(  C.P[k|k-1].(C') + R  )
 22%    xhat[k|k]   = xhat[k|k-1]  +  K[k]*( y[k] - C.xhat[k|k-1]  )
 23%       P[k|k]   = (I-K[k].C). P[k|k-1] . (I-K[k].C)'   +  K[k]R.(K[k]')
 24%                = (I-K[k].C). P[k|k-1]
 25%--------------------------------------------------------------------------
 26% INITIAL conditions:
 27%     x[0] - state value at time ZERO
 28%     P[0] - covariance matrix of the state estimate
 29%--------------------------------------------------------------------------
 30% VECTOR description:
 31%     x[k] - Nx_x_1 , STATE  vector at time k
 32%     u[k] - Nu_x_1 , INPUT  vector at time k
 33%     y[k] - Ny_x_1 , OUTPUT vector at time k
 34% 
 35%     w[k] - Nx_x_1 , PROCESS_NOISE     vector at time k
 36%     v[k] - Ny_x_1 , MEASUREMENT_NOISE vector at time k
 37%--------------------------------------------------------------------------
 38% MATRIX description:
 39%     A    - Nx_x_Nx
 40%     B    - Nx_x_Nu
 41%     C    - Ny_x_Nx
 42% 
 43%     Q    - Nx_x_Nx, PROCESS_NOISE     covariance matrix
 44%     R    - Ny_x_Ny, MEASUREMENT_NOISE covariance matrix
 45%     P    - Nx_x_Nx, covariance matrix of the state estimate
 46%     K    - Nx_x_Ny, Kalman gain matrix
 47%--------------------------------------------------------------------------
 48% EXPECTED_USAGE_#1:
 49%   >> OBJ = my_KF_general_CLS( A, B, C, Q, R, x0, P0 );
 50%   >> OBJ = OBJ.predict(u_k_minus_1);
 51%   >> OBJ = OBJ.correct(y_k);
 52%--------------------------------------------------------------------------
 53% EXPECTED_USAGE_#2:
 54%   >> OBJ = my_KF_general_CLS( A, B, C, Q, R, x0, P0 );
 55%   >> [xo_mat, yo_mat] = OBJ.batch_step(u_mat, y_meas_mat, do_plots);
 56%--------------------------------------------------------------------------
 57
 58properties (SetAccess = protected)
 59    A  = [];
 60    B  = [];
 61    C  = [];
 62    Q  = [];
 63    R  = [];
 64    x0 = [];
 65    P0 = [];
 66    %------------------------------
 67    Nx = 0;
 68    Nu = 0;
 69    Ny = 0;  
 70    %------------------------------
 71    x
 72    P
 73end
 74%--------------------------------------------------------------------------
 75%_#########################################################################
 76methods
 77function OBJ = my_KF_general_CLS( A, B, C, Q, R, x0, P0 )
 78    if(1 ==coder.target('MATLAB') )
 79       validateattributes(A,{'numeric'},{'square'})
 80       validateattributes(Q,{'numeric'},{'square'})
 81       validateattributes(R,{'numeric'},{'square'})
 82       validateattributes(P0,{'numeric'},{'square'})
 83    end
 84    Nx     = size(A,1);
 85    Nu     = size(B,2);
 86    Ny     = size(C,1);
 87       
 88    OBJ.A  = A;
 89    OBJ.B  = B;
 90    OBJ.C  = C;
 91    OBJ.Q  = Q;
 92    OBJ.R  = R;
 93    OBJ.x0 = x0(:);
 94    OBJ.P0 = P0;
 95    
 96    OBJ.Nx = Nx;
 97    OBJ.Ny = Ny;
 98    OBJ.Nu = Nu;  
 99    
100    OBJ.x  = OBJ.x0;
101    OBJ.P  = P0;
102end % my_KF_general_CLS()

103%-------------------------------------------------------------------------- 
104function  OBJ = predict(OBJ, varargin)
105    % ALLOWED USAGE:
106    % >> OBJ = OBJ.predict()
107    % >> OBJ = OBJ.predict(u)
108    
109    if(2==nargin)
110      u = varargin{1};
111      u = u(:);
112      assert( ~isempty(OBJ.B),  '###ERROR:  hey! you gave me a U, but your B is empty !');
113      
114      assert(length(u)==size(OBJ.B,2), '###_ERROR:  hey your u and B have a dimension INconsistency !')
115    end
116    
117    %    xhat[k|k-1] = A.x[k-1]  +  B.u[k-1]
118    %       P[k|k-1] = A.P[k-1|k-1].(A')  +  Q
119    
120    x = OBJ.x;
121    A = OBJ.A;
122    P = OBJ.P;
123    Q = OBJ.Q;
124    if(isempty(OBJ.B))
125        x = A*x;
126    else
127        B = OBJ.B;
128        x = A*x  +  B*u;
129    end
130    
131    OBJ.x = x;
132    OBJ.P = A*P*(A.') + Q;
133end % predict()
134%-------------------------------------------------------------------------- 

135function  OBJ = correct(OBJ, y)
136    if(1 ==coder.target('MATLAB') )
137       validateattributes(y,{'numeric'},{'vector'});
138    end
139    
140    y = y(:);
141    
142    assert(length(y)==size(OBJ.C,1), '###_ERROR:  hey your y and C have a dimension INconsistency !')
143
144    x = OBJ.x; %#ok<*PROPLC>
145    A = OBJ.A; %#ok<*NASGU>
146    C = OBJ.C;
147    P = OBJ.P;
148    Q = OBJ.Q;
149    R = OBJ.R;
150    I = eye(OBJ.Nx);
151    
152    % Recall the BACK and FORWARD slash operators
153    %  A.X=B ---> X=inv(A)*B ---> X = A\B
154    %  X.A=B ---> X=B*inv(A) ---> X = B/A
155    
156    %     K[k]     = P[k|k-1].(C'). inv(  C.P[k|k-1].(C') + R  )
157    %  xhat[k|k]   = xhat[k|k-1]  +  K[k]*( y[k] - C.xhat[k|k-1]  )
158    %     P[k|k]   = (I-K[k].C). P[k|k-1] . (I-K[k].C)'   +  K[k]R.(K[k]')
159    %                = (I-K[k].C). P[k|k-1]
160    
161    K = (P* (C.')) / (C*P*(C.') + R);
162    x = x + K*(y - C*x);
163    P = (I - K*C)*P;
164    
165    OBJ.x = x;
166    OBJ.P = P;
167end % correct()
168%-------------------------------------------------------------------------- 
169function x = get_state(OBJ)
170         x = OBJ.x;
171end % get_state()
172%-------------------------------------------------------------------------- 
173function [xo_mat, yo_mat, varargout] = batch_step(OBJ, u_mat, y_mat, tf_do_plots)
174% ALLOWED_USAGE:
175%  >> [xo_mat, yo_mat]      = batch_step(OBJ, u_mat, y_mat)
176%  >> [xo_mat, yo_mat]      = batch_step(OBJ, u_mat, y_mat, true)
177%  >> [xo_mat, yo_mat, hax] = batch_step(OBJ, u_mat, y_mat, true)    
178%
179% EXPECTED INPUTS: (please see ATTENTION note below)
180%   y_mat = Ny_x_Nsteps
181%   u_mat = Nu_x_(Nsteps-1) or EMPTY
182%-----------------------------------------------------------
183%   ***ATTENTION***:
184%                    ycol_at_time_k         = y_mat(:,k)
185%                    ucol_at_time_k_minus_1 = u_mat(:,k)
186%
187% By packing y_mat and u_mat in this fashion it makes it pretty easy to
188% cycle through each time step within a FOR loop (see below)
189%-----------------------------------------------------------
190% OUTPUTS:
191%   xo_mat =  Nx_x_Nsteps
192%   yo_mat =  Ny_x_Nsteps
193%-----------------------------------------------------------
194      
195    % check the inputs
196    assert(size(y_mat,1)==OBJ.Ny, 'ERR: your y_mat has INcompatable dimensions');
197    assert(size(u_mat,1)==OBJ.Nu, 'ERR: your u_mat has INcompatable dimensions')
198
199    N_steps  = size(y_mat,2);
200    Nx       = OBJ.Nx;
201    Ny       = OBJ.Ny;
202    xo_mat   = zeros(Nx, N_steps);
203    yo_mat   = zeros(Ny, N_steps);
204    
205    % step through each time instance
206    for k=1:N_steps
207       if(isempty(u_mat))
208           u = [];
209       else
210           % this is u_at_k_minus_1
211           u = u_mat(:,k);
212       end
213       
214       % this is y_at_k
215       y   = y_mat(:,k);
216       
217       OBJ = OBJ.predict(u);
218       
219       if(any(isnan(y)))
220           % do NOT correct
221       else
222          OBJ = OBJ.correct(y);
223       end
224       
225       xo_mat(:,k) = OBJ.x;
226       
227       yo_mat(:,k) = OBJ.C*OBJ.x;
228    end % for loop
229    
230    % should we do any plots ?
231    if(nargin==3)
232        % NOPE: do NOT do any plotting
233      return
234    elseif( 4==nargin && false==tf_do_plots)
235        % NOPE: do NOT do any plotting
236        return
237    end
238    
239    % OK: if we made it to here then PLOT stuff
240    hax = [];
241    
242    if(1 ==coder.target('MATLAB') )
243        hax = OBJ.plot(y_mat, yo_mat, xo_mat);
244    end
245    
246    % take care of any OPTIONAL outputs
247    if(3==nargout)
248       varargout{1} = hax; 
249    end
250    
251end % batch_step()
252%--------------------------------------------------------------------------
253function hax = plot(OBJ, y_mat, yo_mat, xo_mat)
254    
255    % check the inputs
256    assert(size(y_mat,1)==OBJ.Ny, 'ERR: your y_mat has INcompatable dimensions');
257   
258    hax      = [];
259    
260    Nx       = OBJ.Nx;
261    Ny       = OBJ.Ny;
262
263    if(1 ==coder.target('MATLAB') )
264           % plot the STATES 
265           figure;
266           hax = [];
267           for kx=1:OBJ.Nx
268                hax(kx) = subplot(Nx,1,kx); 
269                plot(xo_mat(kx,:),'-k','LineWidth',2); grid('on'); axis('tight');
270                tmpstr = sprintf('X(%d) EST',kx);
271                legend({tmpstr},'Location','Best');
272           end
273
274           % plot the OUTPUTS
275           figure;
276           Ny = OBJ.Ny;
277           for ky=1:Ny
278              subplot(Ny,1,ky); 
279                plot( y_mat(ky,:),'-r');  hold on
280                plot(yo_mat(ky,:),'-k','LineWidth',2); 
281                  grid('on'); axis('tight');
282                  tmpCSE{1} = sprintf('Y meas');
283                  tmpCSE{2} = sprintf('Y(%d) EST',ky);
284                  legend(tmpCSE,'Location','Best');
285           end       
286    end % if(1 ==coder.target('MATLAB') )
287end
